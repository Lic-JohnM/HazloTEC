import pandas as pd
import matplotlib.pyplot as plt

# Cargar modelo STL
malla = trimesh.load('modelo.stl')

# Extraer propiedades

    '츼rea superficial (mm)': [area],

# Crear vistas
def guardar_vista(malla, direccion, nombre_archivo):
    escena = malla.scene()
    escena.set_camera(angles=direccion)
    imagen = escena.save_image(resolution=(800, 600), visible=True)
    with open(nombre_archivo, 'wb') as f:
        f.write(imagen)

guardar_vista(malla, (0, 0, 0), 'vista_frontal.png')
guardar_vista(malla, (0, -90, 0), 'vista_lateral.png')
guardar_vista(malla, (90, 0, 0), 'vista_superior.png')
guardar_vista(malla, (45, 45, 45), 'vista_isometrica.png')

plt.text(0.5, 0.4, f"Alto Z: {dimensiones[2]:.1f} mm", ha='center')
